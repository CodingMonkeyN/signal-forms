applyEach()

// Schema for each array item
const lineItemSchema = schema<LineItem>(
  (item) => {{ '{' }}
          required(item.product);
    min(item.quantity, 1);
    min(item.price, 0.01);
          {{ '}' }}
          );

// Apply to each array element
invoiceForm = form(model, (inv) => {{ '{' }}
          applyEach(inv.items, lineItemSchema);
          {{ '}' }});

// Add/remove: just update the model!
model.update(m => ({{ '{' }}
          ...m,
  items: [...m.items, newItem]
          {{ '}' }}));

// Iterate in template
@for (item of form.items; track $index) {{ '{' }}
          <input [formField]="item.product" />
  <input [formField]="item.quantity" />
          {{ '}' }}

Key Takeaways

  • applyEach() applies a schema to every array item
  • To add/remove items, simply update the model signal
  • Each item gets independent validation state
  • Iterate with @for (item of form.items)
  • Access items: form.items[0].product
  • No FormArray, no push(), no removeAt() - just signal updates!
